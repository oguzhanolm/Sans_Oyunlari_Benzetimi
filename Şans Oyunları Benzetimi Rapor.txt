//-------------------------SIRALAMA FONKSIYONU----------------------------| TAMAMLANDI
int siralama(int sirala[],int uzunluk)
{
int i,j,gecici;
  for (i=0; i<uzunluk-1; i++) {
    for (j=0; j<uzunluk-1-i; j++)
      if (sirala[j+1] < sirala[j]) {
        gecici = sirala[j];
        sirala[j] = sirala[j+1];
        sirala[j+1] = gecici;
    }
  }

}
